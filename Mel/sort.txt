
document.querySelector('#sort-asc').onclick = mySort;
document.querySelector('#sort-desc').onclick = mySortDesc;

function mySort() {
    let articles = document.querySelector('#articles');
    for (let i=0; i < articles.children.length; i++) {
        for (let j=i; j < articles.children.length; j++) {
            if (+articles.children[i].getAttribute('data-sort') > +articles.children[j].getAttribute('data-sort')) {
                replacedNode = articles.replaceChild(articles.children[j], articles.children[i]); 
                insertAfter(replacedNode, articles.children[i])
            }
        }
    }
}
